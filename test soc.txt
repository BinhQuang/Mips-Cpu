// $t3: địa chỉ RAM lưu kết quả

ori   $t1, $t1, 0x7F00    // $t1 = 0x00007F00 (base GPIO)

ori   $t0, $t0, 0xAAAA     // $t0 = 0xAAAA
sw    $t0, 0x04($t1)       // Ghi $t0 vào GPIO_OUT tại 0x7F04

sw    $t2, 0x00($t3)       // Ghi giá trị đọc được từ GPIO_IN vào RAM[0x1000]
